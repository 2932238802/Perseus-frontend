#include "mainwindow.h"

/**
 * CppEditor - A simple C++ IDE built with Qt
 * Author: shengjie.lin
 */

int main(int argc, char *argv[])
{
    QApplication app(argc, argv);

    // 设置应用信息
    app.setApplicationName("CppEditor");
    app.setApplicationVersion("1.0.0");
    app.setOrganizationName("LosAngelous");

    MainWindow window;
    window.resize(1280, 800);
    window.setWindowTitle("CppEditor");
    window.show();

    return app.exec();
}
